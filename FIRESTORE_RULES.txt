// Copy and paste these rules into your Firebase Console
// Go to: Firestore Database → Rules tab → Paste this → Publish

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow anyone to read likes
    match /thoughts/likes {
      allow read: if true;

      // Allow anyone to increment likes, but prevent decrements
      allow update: if true;
      allow create: if true;
    }

    // Coffee orders: allow customers to create orders and the public list to read them.
    match /orders/{orderId} {
      allow create: if true;
      allow read: if true;
      allow update, delete: if false;
    }


    // Coffee devblog comments
    match /coffeeComments/{commentId} {
      allow read, write: if true;
    }

    // Menu availability is public read-only from the site. Update it from Firebase Console.
    match /menu/{docId} {
      allow read: if true;
      allow write: if false;
    }

    // Total order counter
    match /stats/{docId} {
      allow read, write: if true;
    }

    // Email list signups
    match /emails/{emailId} {
      allow create: if true;
      allow read, update, delete: if false;
    }

    // Slot counts for limiting orders
    match /orderSlots/{slotId} {
      allow read, write: if true;
    }

    // Deny all other access by default
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
